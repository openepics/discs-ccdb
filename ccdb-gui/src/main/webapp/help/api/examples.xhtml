Examples

The following curl commands retrieve 'events' and 'summary report' resources respectively:

  • curl -v -k -L -H "Accept: application/json" -H "DISCS-Authorization: key:pass" -X GET '#{msgs.apiBaseUrl}/notes?facility=CCF&from=2015-02-08'
  • curl -v -k -L -H "Accept: application/xml" -H "DISCS-Authorization: key:pass" -X GET '#{msgs.apiBaseUrl}/reports/summary'

The following JavaScript code snippet retrieves summary report resource in JSON. 

var request = require('request');

var summary = {};

function getSummary() {
  request({
    url: dataconfig.ccdbreport,
    strictSSL: false,
    headers: {
      Accept: 'application/json',
      'DISCS-Authorization': dataconfig.ccdbkey
    },
    timeout: 30 * 1000
  }, function (error, response, body) {
    if (error) {
      console.error(error);
    } else {
      if (response.statusCode === 200) {
        summary = body;
      }
    }
  });
}

app.get('/device/ACC', function (req, res) {
    getSummary();
    res.type('json');
    res.send(summary);
});
